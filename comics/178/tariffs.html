import * as operations from "Math";
import { PI } from "./constants";
import { fourier, relativity } from "../utils";

export const TWOPI = PI * 2;
export const THREE_PI = PI * 3;
export const double = (a) => operations.double(a);
export const halve = n => n / 2;
export function chaos(a, b) {
  return a > b && Math.random() > 0.5 
    ? fourier(a, b)
    : relativity(b, a);
}
